% Create a \parttitle command
\def\@part[#1]#2{%
  \gdef\parttitle{#1}
  \ifnum \c@secnumdepth >-2\relax
    \refstepcounter{part}%
    \addcontentsline{toc}{part}{\thepart\hspace{1em}#1}%
  \else
    \addcontentsline{toc}{part}{#1}%
  \fi
  \markboth{}{}%
  {\centering
  \interlinepenalty \@M
  \normalfont
  \ifnum \c@secnumdepth >-2\relax
    \huge\bfseries \partname\nobreakspace\thepart
    \par
    \vskip 20\p@
  \fi
  \Huge \bfseries #2\par}%
  \@endpart
}
\def\@spart#1{%
  \gdef\parttitle{#1}
  {\centering
  \interlinepenalty \@M
  \normalfont
  \Huge \bfseries #1\par}%
  \@endpart
}

  \newcommand{\tryincludegraphics}[1]%
